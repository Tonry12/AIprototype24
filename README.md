        '--bd',
        type=parse_date,
        required=True,
        help='Birthday of the user in format DD/MM/YYYY'
    )
    parser.add_argument(
        '--name',
        type=str,
        default='Ratchanont',
        help='Input the name of the person using the app'
    )

    return args

    delta = bd - today
